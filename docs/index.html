A JavaScript library to read, write, and manipulate OpenType fonts.

Load & Measure

Parse any OpenType or TrueType font and access glyph metrics, kerning, and layout features.

// Load and measure a font with opentype.js
import opentype from 'opentype.js';

const buffer = await fetch('fonts/MyFont.ttf').then(r => r.arrayBuffer());
const font = opentype.parse(buffer);

// Draw text on a canvas
const ctx = canvas.getContext('2d');
font.draw(ctx, 'Hello', 0, 120, 72);

// Measure advance width
const width = font.getAdvanceWidth('Hello', 72);

// Access individual glyphs
const glyph = font.charToGlyph('A');
console.log(glyph.name, glyph.advanceWidth);

Manipulate Outlines

Access raw path commands and transform letterforms programmatically.

// Manipulate glyph outlines with opentype.js
import opentype from 'opentype.js';

const buffer = await fetch('fonts/MyFont.ttf').then(r => r.arrayBuffer());
const font = opentype.parse(buffer);
const path = font.getPath('Hello', 0, 150, 72);

// Snap all points to a grid
const snap = (v, grid, strength) =>
    v * (1 - strength) + strength * Math.round(v / grid) * grid;

for (const cmd of path.commands) {
    if (cmd.type !== 'Z') {
        cmd.x = snap(cmd.x, 50, 0.8);
        cmd.y = snap(cmd.y, 50, 0.8);
    }
}

path.draw(ctx); // Draw the modified path
